## Installing miniconda  
mkdir -p ~/miniconda3
wget https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh -O ~/miniconda3/miniconda.sh
bash ~/miniconda3/miniconda.sh -b -u -p ~/miniconda3
rm -rf ~/miniconda3/miniconda.sh
~/miniconda3/bin/conda init bash
~/miniconda3/bin/conda init zsh # close and reopen terminal to make the changes happen

nano ~/.bashrc # open configuration file, scroll to the bottom of the file, then add the following line
export PATH="/temporario2/11217468/miniconda3/bin:$PATH" # setting up the path to be able to use conda commands; Save the file and exit the text editor (in nano, you can do this by pressing Ctrl + X, then Y, then Enter) # close the terminal and reopen 

conda create --name monkey-blast # creating environment for analyzing data 

### in case there's a persistent issue with conda initialization and shell configuration, run the following
source ~/.bashrc
conda init --all
conda activate ~/miniconda3/envs/monkey-blast

### Installing required packages 
conda install bioconda::blast # in the first
